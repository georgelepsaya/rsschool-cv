### Code example
This is my solution of "Multiples of 3 or 5" task from codewars:

  const solution = (num) => {
    const res = [];
    if (num < 0) {
      return 0;
    }
    for (let i = 1; i < num / 5; i++) {
      if (!res.includes(5 * i)) {
        res.push(5 * i);
      }
    }
    for (let i = 1; i < num / 3; i++) {
      if (!res.includes(3 * i)) {
        res.push(3 * i);
      }
    }
    return res.reduce((prev, a) => prev + a, 0);
  };
